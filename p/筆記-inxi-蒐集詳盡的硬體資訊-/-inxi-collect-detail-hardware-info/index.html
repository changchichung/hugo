- name: run inxi to collect Ubuntu/Debian hardware info
  become: yes
  become_method: sudo
  shell: "/usr/local/bin/inxi -c -Dxx -C -m -Z"
  register: du_hw_info
  when: ansible_distribution == "Ubuntu" or ansible_distribution == "Debian" or ansible_distribution == "CentOS"
